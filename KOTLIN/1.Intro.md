##Code 2:

'''fun main()
{
  val name = "World"
  println("Hello "+name)
}'''


<!-- prints Hello World -->
<!-- can also print the same using 'println("Hello $name")' -->
<!-- val is an immutable variable more like constant in other langauges -->
<!-- var is used to define mutable variables -->
<!-- to make the code readable it is good to write the datatype along with definition -->
<!-- ' var name = "World" ' can be written as ' var name: String = "World" ' -->
